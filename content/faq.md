[[faq]]
  question = "Combien d'invités peut accueillir le château ?"
  answer = "Le Château de Laubardemont accueille de 50 à 120 invités pour un dîner assis, en salle voutée ou en extérieur sous l'allée des tilleuls. Pour un cocktail, la capacité peut aller jusqu'à 150 personnes."

[[faq]]
  question = "Combien de couchages sont disponibles sur place ?"
  answer = "Le domaine dispose de 60 couchages répartis entre la suite nuptiale, les chambres de caractère du château (chacune avec salle de bain privative), les maisons de pierre indépendantes et les appartements. Un dortoir enfants est également disponible."

[[faq]]
  question = "Le domaine est-il privatisé pour notre événement ?"
  answer = "Oui, le Château de Laubardemont est entièrement privatisé pour chaque mariage. Vous disposez de la totalité du domaine : château, jardins, terrasse, piscine et dépendances pour tout le week-end."

[[faq]]
  question = "Y a-t-il une limite d'heure pour la musique ?"
  answer = "Non, il n'y a aucune limite d'heure ni en intérieur ni en extérieur. Le château est situé sur une île, à l'écart du voisinage, ce qui permet de profiter de la fête sans contrainte horaire."

[[faq]]
  question = "Peut-on organiser une cérémonie laïque au château ?"
  answer = "Oui, l'allée des tilleuls centenaires forme une nef végétale de cent mètres, idéale pour une cérémonie laïque. D'autres espaces sont possibles : la terrasse, les jardins anglais ou la cour intérieure."

[[faq]]
  question = "Les prestataires sont-ils imposés ?"
  answer = "Non, vous êtes entièrement libres de choisir vos prestataires (traiteur, photographe, fleuriste, DJ…). Nous recommandons des partenaires de confiance qui connaissent le domaine, mais sans aucune obligation."

[[faq]]
  question = "Où se situe le Château de Laubardemont ?"
  answer = "Le château est situé à Sablons (33910), en Gironde, à environ 40 minutes de Bordeaux et 15 minutes de Libourne. Il est facilement accessible par l'autoroute A89."

[[faq]]
  question = "Peut-on visiter le château avant de réserver ?"
  answer = "Oui, nous proposons des visites sur rendez-vous. Contactez-nous pour convenir d'une date. La visite dure environ 1h30 et permet de découvrir l'ensemble du domaine."

[[faq]]
  question = "Un brunch du lendemain est-il possible ?"
  answer = "Oui, c'est même l'un des grands atouts du château. Avec 60 couchages sur place, vos invités se retrouvent le lendemain matin pour un brunch convivial au bord de la rivière ou en terrasse."

[[faq]]
  question = "Quelles sont les dates disponibles ?"
  answer = "Le château accueille des mariages de mars à novembre. Les week-ends d'été se réservent souvent un an à l'avance. Contactez-nous pour vérifier la disponibilité de votre date."
